Task SQL
How many people are in database SELECT COUNT(id) FROM users;
How many people are from New York? SELECT COUNT(id) FROM users WHERE state = "NY";
How many people have opted out in total SELECT COUNT(opted_out) FROM users;
Show each state and the number of people in that state SELECT state, COUNT(state) FROM users GROUP BY state;
How many total cars are owned? SELECT SUM(cars) FROM users;
How many total cars are owned by people in FL? SELECT SUM(cars) FROM users WHERE state = "FL";
What are the most number of cars owned by any one person? SELECT MAX(cars) FROM users;
What are the least number of cars owned by any one person? SELECT MIN(cars) FROM users;
How many cars are owned by each family(by last name) in order of which family has most cars? SELECT lname, SUM(cars) FROM users GROUP BY lname ORDER BY SUM(cars) DESC;
How many cars are owned in each state by Smiths in order of which state had the most cars? SELECT SUM(cars), lname, state FROM users WHERE lname = "Smith" GROUP BY state ORDER BY SUM(cars) DESC;
Show the states where Smiths own more than 2 cars. SELECT state, SUM(cars) FROM users WHERE lname = "Smith" GROUP BY state HAVING SUM(cars) > 2;
